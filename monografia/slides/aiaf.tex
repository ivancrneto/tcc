\frame
{  
  \frametitle{AIAF}
Pilares:
\begin{itemize}
 \item Usabilidade;
 \item Gerenciamento de Informação;
 \item Transparência.
\end{itemize}
}

\frame
{  
  \frametitle{AIAF}
Objetivos:
\begin{itemize}
 \item Extensibilidade;
 \item Facilidade de uso;
 \item Rapidez;
 \item Fácil manutenção;
 \item Integração com código legado.
\end{itemize}
}

\frame
{  
  \frametitle{AIAF}
Casos de uso:
\begin{itemize}
 \item Criar novo projeto;
 \item Abrir projeto existente;
 \item Gerar matriz de adjacências de um limiar qualquer;
 \item Gerar matriz de vizinhança de um limiar qualquer;
 \item Visualizar gráfico de matriz de vizinhança em forma de matriz de cores;
 \item Analisar limiares;
 \item Executar clusterização;
 \item Definir módulos/comunidades;
 \item Comparar/executar congruência de redes de diferentes projetos;
 \item Visualizar gráfico completo da rede;
 \item Executar congruência em relação a um arquivo cadastrado em um formato padrão;
 \item Exportar dados do projeto para gráficos,
\end{itemize}
}

\frame
{  
  \frametitle{AIAF}
Modelagem:
\begin{itemize}
 \item Projeto - classe \textit{Project}
 \begin{itemize}
  \item Eventos da GUI;
  \item Gerenciamento do sistema;
  \item Salvamento de dados - pacote \textit{Persistence}.
\end{itemize}
\end{itemize}
}
